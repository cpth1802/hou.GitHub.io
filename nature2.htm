    // 2. 遊戲狀態變數

    let currentQuiz = []; // 當前 5 題

    let currentQIndex = 0; // 當前題目索引 (0-4)

    let totalScore = 0; // 累積總分

    let totalPoints = 0; // 累積點數

    let userAnswers = {}; // 儲存使用者本輪答案 {q_id: 'A'}

 

    // 3. DOM 元素

    const quizArea = document.getElementById('quiz-area');

    const scoreDisplay = document.getElementById('score');

    const pointsDisplay = document.getElementById('points');

    const rewardPointsDisplay = document.getElementById('reward-points');

    const prevBtn = document.getElementById('prev-btn');

    const nextBtn = document.getElementById('next-btn');

    const submitBtn = document.getElementById('submit-btn');

    const newRoundBtn = document.getElementById('new-round-btn');

    const redeemBtn = document.getElementById('redeem-btn');

    const achievementList = document.getElementById('achievement-list');

 

    // 4. 核心功能

 

    // 從題庫中隨機選取 5 題

    function generateNewQuiz() {

        const shuffled = allQuestions.sort(() => 0.5 - Math.random());

        currentQuiz = shuffled.slice(0, 5);

        currentQIndex = 0;

        userAnswers = {}; // 重置答案

    }

 

    // 渲染當前題目

    function renderQuestion(index) {

        if (currentQuiz.length === 0) return;

 

        const qData = currentQuiz[index];

        const optionsHtml = qData.options.map((opt, i) => {

            const letter = String.fromCharCode(65 + i); // 65 = 'A'

            const isChecked = userAnswers[qData.id] === letter ? ' checked' : '';

            return `<button

                        data-answer="${letter}"

                        data-qid="${qData.id}"

                        onclick="handleAnswerClick(this)"

                        ${isChecked ? 'class="selected"' : ''}>

                        (${letter}) ${opt}

                    </button>`;

        }).join('');

 

        const cardHtml = `

            <div class="question-card" data-qid="${qData.id}">

                <h2>第 ${index + 1} 題 (總共 5 題)</h2>

                <p class="question-text">${qData.q}</p>

                ${qData.image ? `<img src="${qData.image}" alt="題目圖示" style="max-width: 100%; height: auto; margin-bottom: 15px;">` : ''}

                <div class="options">${optionsHtml}</div>

                <div class="explanation" id="exp-${qData.id}"><strong>💡 解題說明:</strong> ${qData.exp}</div>

            </div>

        `;

        quizArea.innerHTML = cardHtml;

        updateNavButtons();

       

        // 如果已經作答，確保選項狀態正確

        if (userAnswers[qData.id]) {

             const selectedBtn = document.querySelector(`.options button[data-answer="${userAnswers[qData.id]}"]`);

             if (selectedBtn) selectedBtn.classList.add('selected');

        }

    }

 

    // 處理選項點擊

    function handleAnswerClick(button) {

        const qid = button.dataset.qid;

        const answer = button.dataset.answer;

 

        // 移除同一個問題的其他選項的 selected class

        document.querySelectorAll(`.options button[data-qid="${qid}"]`).forEach(btn => {

            btn.classList.remove('selected');

        });

 

        // 設置當前選項

        button.classList.add('selected');

        userAnswers[qid] = answer;

    }

 

    // 更新導航按鈕狀態

    function updateNavButtons() {

        prevBtn.style.display = currentQIndex > 0 ? 'inline-block' : 'none';

        nextBtn.style.display = currentQIndex < 4 ? 'inline-block' : 'none';

        submitBtn.style.display = currentQIndex === 4 ? 'inline-block' : 'none';

        newRoundBtn.style.display = 'none'; // 預設隱藏

    }

 

    // 檢查並計算分數

    function checkQuiz() {

        let correctCount = 0;

       

        currentQuiz.forEach(q => {

            const userAnswer = userAnswers[q.id];

            const isCorrect = userAnswer === q.ans;

           

            if (isCorrect) {

                correctCount++;

            }

 

            // 顯示解題說明和顏色標註

            const explanationElement = document.getElementById(`exp-${q.id}`);

            if (explanationElement) explanationElement.style.display = 'block';

 

            document.querySelectorAll(`.options button[data-qid="${q.id}"]`).forEach(btn => {

                btn.disabled = true; // 鎖定所有按鈕

                const btnAnswer = btn.dataset.answer;

               

                if (btnAnswer === q.ans) {

                    btn.classList.add('correct');

                } else if (btnAnswer === userAnswer && !isCorrect) {

                    btn.classList.add('wrong');

                }

            });

        });

 

        const pointsGained = correctCount * 5; // 每題 5 點

       

        // 更新分數和點數

        scoreDisplay.textContent = correctCount;

        totalPoints += pointsGained;

        pointsDisplay.textContent = totalPoints;

        rewardPointsDisplay.textContent = totalPoints;

 

        // 顯示總結按鈕，隱藏其他

        prevBtn.style.display = 'none';

        nextBtn.style.display = 'none';

        submitBtn.style.display = 'none';

        newRoundBtn.style.display = 'inline-block';

        

        alert(`本輪挑戰結束！你答對了 ${correctCount} 題，獲得 ${pointsGained} 點數！🎉`);

       

        // 檢查並兌換獎勵

        checkRedeemableRewards();

    }

 

    // 兌換獎勵 (自動或手動)

    function checkRedeemableRewards() {

        const rewards = [

            { id: 'medal-1', cost: 10 },

            { id: 'medal-2', cost: 30 },

            { id: 'medal-3', cost: 50 },

            { id: 'medal-4', cost: 100 }

        ];

 

        rewards.forEach(reward => {

            const medalElement = document.getElementById(reward.id);

            if (totalPoints >= reward.cost && !medalElement.classList.contains('unlocked')) {

                medalElement.classList.add('unlocked');

                medalElement.title = `${medalElement.alt} (已解鎖)`;

                console.log(`恭喜解鎖：${medalElement.alt}`);

            }

        });

    }

 

    // 5. 事件監聽

 

    // 導航按鈕事件

    prevBtn.addEventListener('click', () => {

        if (currentQIndex > 0) {

            currentQIndex--;

            renderQuestion(currentQIndex);

        }

    });

 

    nextBtn.addEventListener('click', () => {

        if (currentQIndex < 4) {

            currentQIndex++;

            renderQuestion(currentQIndex);

        }

    });

 

    // 提交按鈕事件

    submitBtn.addEventListener('click', () => {

        // 先確保所有題目都有作答 (可選，這裡簡化為直接提交)

        checkQuiz();

    });

 

    // 開始新一輪事件

    newRoundBtn.addEventListener('click', () => {

        scoreDisplay.textContent = "0"; // 歸零單輪分數

        startQuiz();

    });

   

    // 獎勵兌換按鈕 (這裡設置為自動檢查，按鈕只是提示作用)

    redeemBtn.addEventListener('click', () => {

        checkRedeemableRewards();

        alert('系統已自動為您檢查可兌換的獎勵！');

    });

 

 

    // 6. 啟動測驗

    function startQuiz() {

        generateNewQuiz();

        renderQuestion(currentQIndex);

        // 重置分數顯示

        scoreDisplay.textContent = '0 / 5';

        updateNavButtons();

    }

 

    // 網頁載入後啟動

    window.onload = startQuiz;
